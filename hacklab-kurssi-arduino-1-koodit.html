void setup() {
Serial.begin(115200);

pinMode(11, OUTPUT);
pinMode(10, OUTPUT);
pinMode(9, OUTPUT);

pinMode(7, INPUT_PULLUP);
}

void loop() {
// Lukee jännitteen 0..5V pinnistä A0
int punainen = analogRead(A0)/4;
int vihrea = analogRead(A1)/4;
int sininen = digitalRead(7);

// Tulostetaan lukuna, omille riveilleen
//Serial.println(arvo);

analogWrite(11, 255 - punainen);
analogWrite(10, 255 - vihrea);
digitalWrite(9, 1 - sininen);

// Pidetään pieni tauko, että serial monitor pysyy perässä
//delay(1);
}
