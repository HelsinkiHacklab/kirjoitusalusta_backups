Kiinalevyn ufc

# board-kiina-xc6slx9-tqg144-2.ucf
#
# Board: No name China
# Family: Spartan-6
# Device: LX9
# Package: TQG144
# Speed: -2

# External oscillator
NET CLOCK_50MHz                   LOC = P21  | IOSTANDARD = LVCMOS33 | PERIOD = 50 MHz; # = 20 ns;

NET LED[0]                        LOC = P44 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[1]                        LOC = P45 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[2]                        LOC = P46 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[3]                        LOC = P47 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[4]                        LOC = P57 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[5]                        LOC = P58 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[6]                        LOC = P59 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;
NET LED[7]                        LOC = P61 | IOSTANDARD = LVCMOS33 | DRIVE=8 | SLEW=SLOW;




// main.v

`timescale 1ns / 1ps
`default_nettype none


module Main #(
parameter                        DIVIDER = 25_000_000,
parameter                        DIVIDER_BITS = $clog2(DIVIDER)
) (
input    wire                     CLOCK_50MHz,
output  wire    [7:0]             LED
);

wire clock = CLOCK_50MHz;

PWM pwm0(.clock(clock), .value(0), .out(LED[0]));
PWM pwm1(.clock(clock), .value(4), .out(LED[1]));
PWM pwm2(.clock(clock), .value(17), .out(LED[2]));
PWM pwm3(.clock(clock), .value(39), .out(LED[3]));
PWM pwm4(.clock(clock), .value(72), .out(LED[4]));
PWM pwm5(.clock(clock), .value(119), .out(LED[5]));
PWM pwm6(.clock(clock), .value(179), .out(LED[6]));
PWM pwm7(.clock(clock), .value(255), .out(LED[7]));

endmodule
