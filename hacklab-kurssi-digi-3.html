// main.v

`timescale 1ns / 1ps
`default_nettype none



// DFF with synchronous reset and clock enable
module DFF(
input   wire        clock,
input   wire        reset,
input   wire        enable,
input   wire        d,
output  reg         q
);

initial             q = 1'b0;

always @(posedge clock) begin
if (reset) begin
q <= 1'b0;
end else if (enable) begin
q <= d;
end
end
endmodule


module main #(
parameter                        EXTERNAL_CLOCK_FREQUENCY_HZ = 100_000_000,
parameter                        EXTERNAL_CLOCK_PERIOD_NS = 1000000000.0/EXTERNAL_CLOCK_FREQUENCY_HZ
) (
input    wire                     CLOCK,
output  wire    [7:0]             LED
);


wire                        clock = CLOCK;
reg    [$clog2(EXTERNAL_CLOCK_FREQUENCY_HZ)-1:0] second_counter = 1'd0;
reg    second_strobe = 1'b0;

always @(posedge clock) begin
if (second_counter != EXTERNAL_CLOCK_FREQUENCY_HZ) begin
second_counter <= second_counter + 1'd1;
second_strobe <= 1'b0;
end else begin
second_counter <= 1'd0;
second_strobe <= 1'b1;
end
end

/*
wire            dff_q;
DFF        dff(
.clock(clock),
.reset(1'b0),
.enable(second_strobe),
.d(~dff_q),
.q(dff_q)
);
*/
wire            dff_q;
FDRE dff(
.C(clock),          // 1-bit Clock input
.R(1'b0),              // 1-bit Synchronous reset input
.CE(second_strobe), // 1-bit Clock enable input
.D(~dff_q),           // 1-bit Data input
.Q(dff_q)          // 1-bit Data output
);


assign LED = { 8{ dff_q } };

endmodule

// main.v

`timescale 1ns / 1ps
`default_nettype none

module main #(
parameter                        EXTERNAL_CLOCK_FREQUENCY_HZ = 100_000_000,
parameter                        EXTERNAL_CLOCK_PERIOD_NS = 1000000000.0/EXTERNAL_CLOCK_FREQUENCY_HZ
) (
input    wire                     CLOCK,
output  wire    [7:0]             LED
);


wire                        clock = CLOCK;
reg    [$clog2(EXTERNAL_CLOCK_FREQUENCY_HZ)-1:0] second_counter = 1'd0;
reg    second_strobe = 1'b0;

always @(posedge clock) begin
if (second_counter != EXTERNAL_CLOCK_FREQUENCY_HZ) begin
second_counter <= second_counter + 1'd1;
second_strobe <= 1'b0;
end else begin
second_counter <= 1'd0;
second_strobe <= 1'b1;
end
end

reg        [7:0]    led = 8'b00000000;

always @(posedge clock) begin
if (second_strobe) begin
led = ~led;
end
end

assign LED = led;

endmodule
