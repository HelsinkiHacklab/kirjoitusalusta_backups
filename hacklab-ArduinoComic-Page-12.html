Panel 47:
 Drawing: code for switch script written on white shape
void setup() {                
  pinMode(2, OUTPUT); 
  pinMode(4, INPUT); 
}

void loop() {
  if(digitalRead(4)){
    digitalWrite(2, HIGH);  
  }else{            
  digitalWrite(2, LOW);   
  }
}
