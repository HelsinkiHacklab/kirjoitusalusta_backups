module Blinky(
input     wire         clk,
output   wire   [3:0] led_a,
output reg [3:0] led_b
);

reg [27:0] count = 0;

// Kombinaatiologiikka
assign led_a = ~count[27:24];

// Tai vaihtoehtoisesti
always @( * ) begin
led_b = ~count[27:24];
end

// Sekvenssilogiikka
always @ ( posedge clk) begin
count <= count + 1;
end

endmodule
