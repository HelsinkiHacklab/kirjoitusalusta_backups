// Verilog harjoitus v1.0

`timescale 1ns / 1ps
`default_nettype none


module main(
input   wire    sw0,
input   wire    sw1,

output  wire    led0
);

wire    sw0_not;
wire    sw1_not;

assign sw0_not = ~sw0;
assign sw1_not = ~sw1;

assign led0 = sw0_not & sw1_not;

endmodule
