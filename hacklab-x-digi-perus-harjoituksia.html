`default_nettype none

module blinky2(
input wire clk,

);

reg [27:0] count = 0;

assign led = ~count[27:24];

always @ (posedge clk) begin
count <= count + 1;
end

// Binäärilaskuri ledeillä, versio 2
// vapaasti säädettävä taajuus

`timescale 1ns / 1ps
`default_nettype none

module blinky2(
input wire clk,
output wire [3:0] led
);

reg [25:0] prescaler = 0;
wire tick;

assign tick = (prescaler == 26'd50000000);

always @ (posedge clk) begin
if (tick) begin
prescaler <= 0;
end
else begin
prescaler <= prescaler + 1;
end
end

reg [3:0] count = 0;

assign led = ~count;

always @ (posedge clk) begin
if (tick) begin
count <= count + 1;
end
end

endmodule
