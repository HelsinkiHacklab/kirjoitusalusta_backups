$fs = 0.1;
$fa = 1;

module decoration1()
{
cube(10,center=true);
}

module ring(r0, r1, r2, h, dn)
{
difference() {
union() {
cylinder(r=r1,h=h,center=true);
intersection() {
for(i=[0:dn-1]) {
rotate([0,0,360/dn*i]) translate([r1,0,0]) rotate([0,90,0]) decoration1();
};
cylinder(r=r2,h=h,center=true);
}
}
cylinder(r=r0,h=h+1,center=true);
}

}

ring(35, 40, 45, 20, 9);
